## 1. 内置四种方法

```js
localStorage.setItem(key, value);

localStorage.getItem(key);

localStorage.removeItem(key);

localStorage.clear();
```

## 2. 封装

```js
var storage = {
  set(key, value) {
    localStorage.setItem(key, JSON.stringify(value));
  },
  get(key) {
    return JSON.parse(localStorage.getItem(key));
  },
  remove(key) {
    localStorage.removeItem(key);
  },
};

export default storage;
```
